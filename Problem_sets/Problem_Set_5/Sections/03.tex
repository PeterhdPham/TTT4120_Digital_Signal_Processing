\subsection*{Find the transfer function $H(z)$ for the filter in figure \ref{fig:echo_generator.png}. }

$$
\begin{aligned}
Y(z) & =X(z)+\alpha z^{-R} X(z) \\
& =X(z)\left(1+\alpha z^{-R}\right) \\
H(z) & =\frac{Y(z)}{X(z)}=1+\alpha z^{-R}
\end{aligned}
$$
